
void pg_lcd_prints_color(const char *str, char color) {
    while (*str) {
        if (*str == '\n') {
            st7789_col = 0;
            st7789_row = (st7789_row + 1) % 15;
        }
        else if (*str == '\r') {
            st7789_col = 0;
        }
        else {
            pg_lcd_draw_char(st7789_col << 4, st7789_row << 4, *str, color, 1);
            _pg_lcd_update_pos();
        }
        str++;
    }
}
